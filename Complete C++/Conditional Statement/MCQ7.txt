In C++, what statement do you use to check whether a variable A is equal to 2? And if it is, print 9. Select one of the following options.


1.  if(A == 2); {
        cout << 9;
    }

2.  if(A == 2){
        cout << 9;
    }
 
3.  if(A == 2){
        cout << 9;
    };

4.  if(A == 2){
        printf("9");
    }




Answer: 
2.  if(A == 2){
        cout << 9;
    }
